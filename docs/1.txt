// app/Queries/Analytics/BaseAnalyticsQuery.php
abstract class BaseAnalyticsQuery
{
    protected $cache = [];
    protected $cacheTtl = 3600; // ساعة
    
    abstract public function get($branchId = null, $period = null);
    
    protected function cacheKey($key)
    {
        return "analytics." . class_basename($this) . "." . $key;
    }
    
    protected function remember($key, $callback)
    {
        return Cache::remember($this->cacheKey($key), $this->cacheTtl, $callback);
    }
}

// app/Queries/Analytics/AttendanceSummaryQuery.php
class AttendanceSummaryQuery extends BaseAnalyticsQuery
{
    public function get($branchId = null, $month = null)
    {
        $key = $branchId . '_' . ($month ?? now()->month);
        
        return $this->remember($key, function() use ($branchId, $month) {
            $month = $month ?? now()->month;
            
            return DB::table('attendance_logs')
                ->select([
                    'branch_id',
                    DB::raw('COUNT(DISTINCT user_id) as total_employees'),
                    DB::raw('SUM(CASE WHEN status = "on_time" THEN 1 ELSE 0 END) as on_time'),
                    DB::raw('SUM(CASE WHEN status = "late" THEN 1 ELSE 0 END) as late'),
                    DB::raw('SUM(CASE WHEN status = "absent" THEN 1 ELSE 0 END) as absent'),
                    DB::raw('SUM(cost) as total_cost'),
                    DB::raw('AVG(TIMESTAMPDIFF(MINUTE, check_in, check_out)) as avg_duration')
                ])
                ->when($branchId, fn($q) => $q->where('branch_id', $branchId))
                ->whereMonth('created_at', $month)
                ->groupBy('branch_id')
                ->get();
        });
    }
}

// app/Queries/Analytics/BranchPerformanceQuery.php
class BranchPerformanceQuery extends BaseAnalyticsQuery
{
    public function get($branchId = null, $year = null)
    {
        $year = $year ?? now()->year;
        $key = ($branchId ?? 'all') . '_' . $year;
        
        return $this->remember($key, function() use ($branchId, $year) {
            $query = DB::table('attendance_logs')
                ->select([
                    'branch_id',
                    DB::raw('YEAR(created_at) as year'),
                    DB::raw('MONTH(created_at) as month'),
                    DB::raw('COUNT(*) as total_logs'),
                    DB::raw('SUM(cost) as monthly_cost')
                ])
                ->whereYear('created_at', $year)
                ->groupBy('branch_id', DB::raw('YEAR(created_at)'), DB::raw('MONTH(created_at)'));
            
            if ($branchId) {
                $query->where('branch_id', $branchId);
            }
            
            return $query->get();
        });
    }

class SendCircularJob implements ShouldQueue

    public function handle()
    {
        $circular = $this->circular;

        $query = User::query()->where('is_active', true);

## 1. نظام Streak متكامل مع العطل والإجازات
```php
// app/Services/StreakCalculator.php
class StreakCalculator
{
    protected $holidayService;
    protected $leaveService;
    
    public function __construct(
        HolidayService $holidayService,
        LeaveService $leaveService
    ) {
        $this->holidayService = $holidayService;
        $this->leaveService = $leaveService;
    }
    
    public function calculate(User $user, Carbon $endDate = null): StreakResult
    {
        $endDate = $endDate ?? now();
        $streak = 0;
        $currentDate = $endDate->copy();
        $workingDaysOnly = []; // أيام العمل الفعلية
        
        while (true) {
            // التحقق من العطل والإجازات
            if ($this->holidayService->isHoliday($currentDate)) {
                $workingDaysOnly[] = [
                    'date' => $currentDate->copy(),
                    'type' => 'holiday',
                    'counts' => false
                ];
                $currentDate->subDay();
                continue;
            }
            
            if ($this->leaveService->isOnApprovedLeave($user, $currentDate)) {
                $workingDaysOnly[] = [
                    'date' => $currentDate->copy(),
                    'type' => 'leave',
                    'leave_type' => $this->leaveService->getLeaveType($user, $currentDate),
                    'counts' => false
                ];
                $currentDate->subDay();
                continue;
            }
            
            // التحقق من الحضور
            $attendance = AttendanceLog::where('user_id', $user->id)
                ->whereDate('check_in', $currentDate)
                ->whereIn('status', ['on_time', 'late'])
                ->first();
            
            if (!$attendance) {
                break;
            }
            
            $workingDaysOnly[] = [
                'date' => $currentDate->copy(),
                'type' => 'workday',
                'status' => $attendance->status,
                'counts' => true
            ];
            
            $streak++;
            $currentDate->subDay();
        }
        
        return new StreakResult(
            streak: $streak,
            history: array_reverse($workingDaysOnly),
            startDate: $currentDate->addDay(),
            endDate: $endDate
        );
    }
}

// app/ValueObjects/StreakResult.php
class StreakResult
{
    public function __construct(
        public int $streak,
        public array $history,
        public Carbon $startDate,
        public Carbon $endDate
    ) {}
    
    public function toArray(): array
    {
        return [
            'streak' => $this->streak,
            'start_date' => $this->startDate->format('Y-m-d'),
            'end_date' => $this->endDate->format('Y-m-d'),
            'working_days' => collect($this->history)->where('counts', true)->count(),
            'excluded_days' => collect($this->history)->where('counts', false)->count(),
            'history' => $this->history
        ];
    }
}

// app/Services/HolidayService.php
class HolidayService
{
    protected $cache = [];
    
    public function isHoliday(Carbon $date): bool
    {
        $key = $date->format('Y-m-d');
        
        if (!isset($this->cache[$key])) {
            $this->cache[$key] = OfficialHoliday::whereDate('date', $date)->exists()
                || $this->isWeekend($date);
        }
        
        return $this->cache[$key];
    }
    
    protected function isWeekend(Carbon $date): bool
    {
        // يمكن جعل أيام العطل الأسبوعية قابلة للتكوين
        return in_array($date->dayOfWeek, [Carbon::FRIDAY, Carbon::SATURDAY]);
    }
}

// app/Models/User.php
protected static function booted()
{
    static::creating(function ($user) {
        if (empty($user->employee_id)) {
            $user->employee_id = static::generateUniqueEmployeeId();
        }
    });
}

protected static function generateUniqueEmployeeId(): string
{
    do {
        // استخدام التاريخ + رقم عشوائي + حرف تحقق
        $date = now()->format('ymd');
        $random = str_pad(random_int(0, 9999), 4, '0', STR_PAD_LEFT);
        $checkDigit = static::calculateCheckDigit($date . $random);
        $employeeId = "EMP-{$date}-{$random}-{$checkDigit}";
    } while (static::where('employee_id', $employeeId)->exists());
    
    return $employeeId;
}

protected static function calculateCheckDigit(string $base): string
{
    // خوارزمية بسيطة للتحقق (Luhn-like)
    $sum = 0;
    for ($i = 0; $i < strlen($base); $i++) {
        $digit = (int) $base[$i];
        if ($i % 2 == 0) {
            $digit *= 2;
            if ($digit > 9) $digit -= 9;
        }
        $sum += $digit;
    }
    return (string) ((10 - ($sum % 10)) % 10);
}

```

## 1. محرك تعبيرات رياضي آمن (بدون eval)
```php
// app/Services/SafeExpressionEngine.php
class SafeExpressionEngine
{
    protected $allowedFunctions = [
        'sum', 'avg', 'count', 'min', 'max', 'round',
        'abs', 'ceil', 'floor', 'sqrt', 'power'
    ];
    
    protected $allowedOperators = ['+', '-', '*', '/', '(', ')', ','];
    
    protected $maxLength = 500;
    protected $maxNesting = 10;
    
    public function evaluate(string $expression, array $data = [])
    {
        // 1. التحقق من الطول
        if (strlen($expression) > $this->maxLength) {
            throw new ExpressionTooLongException();
        }
        
        // 2. تنظيف التعبير
        $clean = $this->sanitize($expression);
        
        // 3. التحقق من الصيغة
        $this->validateSyntax($clean);
        
        // 4. تحويل إلى RPN (Reverse Polish Notation)
        $rpn = $this->toRPN($clean);
        
        // 5. تقييم RPN بأمان
        return $this->evaluateRPN($rpn, $data);
    }
    
    protected function sanitize(string $expr): string
    {
        // إزالة أي أحرف غير مسموحة
        $expr = preg_replace('/[^0-9\s\+\-\*\/\(\)\.\,\w]/', '', $expr);
        
        // استبدال أسماء الحقول بقيمها
        return $expr;
    }
    
    protected function validateSyntax(string $expr): void
    {
        // التحقق من توازن الأقواس
        $balance = 0;
        $nesting = 0;
        
        for ($i = 0; $i < strlen($expr); $i++) {
            if ($expr[$i] === '(') {
                $balance++;
                $nesting++;
                if ($nesting > $this->maxNesting) {
                    throw new ExpressionTooComplexException();
                }
            } elseif ($expr[$i] === ')') {
                $balance--;
                $nesting--;
            }
        }
        
        if ($balance !== 0) {
            throw new UnbalancedParenthesesException();
        }
    }
    
    protected function toRPN(string $expr): array
    {
        // Shunting-yard algorithm
        $output = [];
        $operators = [];
        $precedence = ['+' => 1, '-' => 1, '*' => 2, '/' => 2];
        
        $tokens = token_get_all('<?php ' . $expr);
        
        foreach ($tokens as $token) {
            if (is_array($token)) {
                switch ($token[0]) {
                    case T_LNUMBER:
                    case T_DNUMBER:
                        $output[] = ['type' => 'number', 'value' => $token[1]];
                        break;
                    case T_STRING:
                        if (in_array($token[1], $this->allowedFunctions)) {
                            $operators[] = ['type' => 'function', 'name' => $token[1]];
                        }
                        break;
                }
            } else {
                // Operators and parentheses
                if (in_array($token, array_keys($precedence))) {
                    while (!empty($operators) && 
                           end($operators)['type'] === 'operator' &&
                           $precedence[end($operators)['value']] >= $precedence[$token]) {
                        $output[] = array_pop($operators);
                    }
                    $operators[] = ['type' => 'operator', 'value' => $token];
                } elseif ($token === '(') {
                    $operators[] = ['type' => 'paren', 'value' => '('];
                } elseif ($token === ')') {
                    while (!empty($operators) && end($operators)['value'] !== '(') {
                        $output[] = array_pop($operators);
                    }
                    array_pop($operators); // Remove '('
                    
                    // Check for function
                    if (!empty($operators) && end($operators)['type'] === 'function') {
                        $output[] = array_pop($operators);
                    }
                }
            }
        }
        
        while (!empty($operators)) {
            $output[] = array_pop($operators);
        }
        
        return $output;
    }
    
    protected function evaluateRPN(array $rpn, array $data): float
    {
        $stack = [];
        
        foreach ($rpn as $token) {
            switch ($token['type']) {
                case 'number':
                    $stack[] = (float) $token['value'];
                    break;
                    
                case 'operator':
                    $b = array_pop($stack);
                    $a = array_pop($stack);
                    
                    switch ($token['value']) {
                        case '+': $stack[] = $a + $b; break;
                        case '-': $stack[] = $a - $b; break;
                        case '*': $stack[] = $a * $b; break;
                        case '/': 
                            if ($b == 0) throw new DivisionByZeroException();
                            $stack[] = $a / $b; 
                            break;
                    }
                    break;
                    
                case 'function':
                    $args = [];
                    for ($i = 0; $i < $token['arg_count'] ?? 1; $i++) {
                        $args[] = array_pop($stack);
                    }
                    $args = array_reverse($args);
                    
                    $stack[] = $this->callFunction($token['name'], $args, $data);
                    break;
            }
        }
        
        return $stack[0] ?? 0;
    }
    
    protected function callFunction(string $name, array $args, array $data): float
    {
        return match($name) {
            'sum' => array_sum($args),
            'avg' => array_sum($args) / count($args),
            'count' => count($args),
            'min' => min($args),
            'max' => max($args),
            'round' => round($args[0], $args[1] ?? 0),
            'abs' => abs($args[0]),
            'ceil' => ceil($args[0]),
            'floor' => floor($args[0]),
            'sqrt' => sqrt($args[0]),
            'power' => pow($args[0], $args[1] ?? 1),
            default => 0
        };
    }
}
```

class ReportFormula extends Model

// tests/Unit/SafeExpressionEngineTest.php
class SafeExpressionEngineTest extends TestCase
{
    protected SafeExpressionEngine $engine;
    
    protected function setUp(): void
    {
        parent::setUp();
        $this->engine = app(SafeExpressionEngine::class);
    }
    
    /** @test */
    public function it_evaluates_basic_arithmetic()
    {
        $this->assertEquals(10, $this->engine->evaluate('5 + 5'));
        $this->assertEquals(15, $this->engine->evaluate('3 * 5'));
        $this->assertEquals(2, $this->engine->evaluate('10 / 5'));
    }
    
    /** @test */
    public function it_rejects_malicious_code()
    {
        $this->expectException(InvalidExpressionException::class);
        $this->engine->evaluate('5 + system("rm -rf /")');
    }
    
    /** @test */
    public function it_handles_variables()
    {
        $data = ['cost' => 100, 'hours' => 8];
        $this->assertEquals(800, $this->engine->evaluate('cost * hours', $data));
    }
    
    /** @test */
    public function it_prevents_division_by_zero()
    {
        $this->expectException(DivisionByZeroException::class);
        $this->engine->evaluate('10 / 0');
    }
    
    /** @test */
    public function it_limits_expression_length()
    {
        $longExpr = str_repeat('1+', 300) . '1';
        $this->expectException(ExpressionTooLongException::class);
        $this->engine->evaluate($longExpr);
    }
}
